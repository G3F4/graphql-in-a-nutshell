DIRECTIVES

                    
query (
  $noName: Boolean
  $logged: Boolean
) {
    user {
        id
        name @skip(
          if: $noName
        )
        loggedIn $include(
          if: $logged
        )
    }
}
JSON: 
                        
{
    "noName": true,
    "logged": true
}
                        
                    
                    
{
    "data": {
        "user" {
            "id": "1",
            "loggedIn": true
        }
    }
}

Type system

    SCALARS

    ONE
    TWO
    THREE

    LISTS

      id: String!

Type system

    INTERFACES

    id: ID!

    id: ID!
    nick: String!

    UNIONS
